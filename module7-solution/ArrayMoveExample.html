let buy = [ {name: "pancakes", quantity: 8, pricePerItem: 8}, {name: "oranges", quantity: 3, pricePerItem: 4}, {name: "cucumbers", quantity: 2, pricePerItem: 3}, {name: "potatoes", quantity: 4, pricePerItem: 6}, {name: "thin mints", quantity: 10, pricePerItem: 20}, {name: "avocados", quantity: 7, pricePerItem: 15} ]; let bought = []; // Function to move an item from 'buy' to 'bought' function moveItem(itemName) { // Find the index of the item to move const itemIndex = buy.findIndex(item => item.name === itemName); // If the item exists, move it to the 'bought' array if (itemIndex !== -1) { // Remove the item from the 'buy' array and add it to the 'bought' array const item = buy.splice(itemIndex, 1)[0]; // splice removes the item bought.push(item); // push adds the item to the 'bought' array } } // Example: Move 'pancakes' from 'buy' to 'bought' moveItem('pancakes'); console.log('Buy:', buy); console.log('Bought:', bought);
